import 'dart:convert';
import 'package:flutter/foundation.dart';
import 'package:http/http.dart' as http;

class OpenIMChatService {
  OpenIMChatService._();
  static final OpenIMChatService instance = OpenIMChatService._();

  static const _chatBase = 'http://34.63.32.143:10008';

  /// Registers a new user. Uses the super-code "666666" so no SMS is needed.
  Future<OpenIMRegisterResult> registerWithPhone({
    required String areaCode,
    required String phoneNumber,
    required String nickname,
    int platform = 2,
  }) async {
    final uri = Uri.parse('$_chatBase/account/register');

    final response = await http
        .post(
          uri,
          headers: {
            'Content-Type': 'application/json',
            'operationID': _opId(),
          },
          body: jsonEncode({
            'verifyCode': '666666',
            'platform': platform,
            'autoLogin': true,
            'user': {
              'nickname': nickname,
              'faceURL': '',
              'areaCode': areaCode,
              'phoneNumber': phoneNumber,
            },
          }),
        )
        .timeout(const Duration(seconds: 15));

    final body = jsonDecode(response.body) as Map<String, dynamic>;
    debugPrint('[OpenIMChat] /account/register response: $body');

    final errCode = body['errCode'] as int? ?? -1;
    if (errCode != 0) {
      final errMsg = body['errMsg'] as String? ?? 'Registration failed';
      throw OpenIMChatException('Registration failed ($errCode): $errMsg');
    }

    final data = body['data'] as Map<String, dynamic>? ?? {};
    return OpenIMRegisterResult(
      userID: data['userID'] as String? ?? '',
      imToken: data['imToken'] as String? ?? '',
      chatToken: data['chatToken'] as String? ?? '',
    );
  }

  /// Logs in an existing user. Uses the super-code "666666".
  Future<OpenIMRegisterResult> loginWithPhone({
    required String areaCode,
    required String phoneNumber,
    int platform = 2,
  }) async {
    final uri = Uri.parse('$_chatBase/account/login');

    final response = await http
        .post(
          uri,
          headers: {
            'Content-Type': 'application/json',
            'operationID': _opId(),
          },
          body: jsonEncode({
            'areaCode': areaCode,
            'phoneNumber': phoneNumber,
            'verifyCode': '666666',
            'platform': platform,
          }),
        )
        .timeout(const Duration(seconds: 15));

    final body = jsonDecode(response.body) as Map<String, dynamic>;
    debugPrint('[OpenIMChat] /account/login response: $body');

    final errCode = body['errCode'] as int? ?? -1;
    if (errCode != 0) {
      final errMsg = body['errMsg'] as String? ?? 'Login failed';
      throw OpenIMChatException('Login failed ($errCode): $errMsg');
    }

    final data = body['data'] as Map<String, dynamic>? ?? {};
    return OpenIMRegisterResult(
      userID: data['userID'] as String? ?? '',
      imToken: data['imToken'] as String? ?? '',
      chatToken: data['chatToken'] as String? ?? '',
    );
  }

  String _opId() => DateTime.now().millisecondsSinceEpoch.toString();

  /// Splits a full phone string into areaCode + phoneNumber.
  static ({String areaCode, String phoneNumber})? splitPhone(
    String fullPhone,
    String dialCode,
  ) {
    final cleaned = fullPhone.replaceAll(RegExp(r'[^\d]'), '');
    final codeDigits = dialCode.replaceAll(RegExp(r'[^\d]'), '');

    if (cleaned.startsWith(codeDigits)) {
      return (
        areaCode: codeDigits,
        phoneNumber: cleaned.substring(codeDigits.length),
      );
    }
    return null;
  }
}

class OpenIMRegisterResult {
  final String userID;
  final String imToken;
  final String chatToken;

  const OpenIMRegisterResult({
    required this.userID,
    required this.imToken,
    required this.chatToken,
  });
}

class OpenIMChatException implements Exception {
  final String message;
  const OpenIMChatException(this.message);

  @override
  String toString() => message;
}
